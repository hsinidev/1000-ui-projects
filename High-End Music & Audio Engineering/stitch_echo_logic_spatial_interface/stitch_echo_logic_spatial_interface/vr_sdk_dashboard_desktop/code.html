void OnSpatialSync() {

  SpatialHook *hook = new SpatialHook(HOOK_TYPE_AUDIO);

  hook->setLatencyThreshold(12.5f);
  hook->enableRealTimeTracking(true);

  if (hook->status() == STATUS_READY) {
    Logger::info("Spatial core initialised successfully.");
    Core::Register(hook);
  } else {
    Logger::critical("Failed to establish spatial handshake.");
  }
}
