const provenance = require('@provenance-chain/sdk');

async function verifyArtwork(artId) {
  const client = new provenance.Client({
    apiKey: 'YOUR_API_KEY',
    environment: 'production'
  });

  try {
    const record = await client.records.fetch(artId);
    return record.is_authentic;
  } catch (err) {
    console.error('VERIFICATION_FAILED', err);
  }
}
