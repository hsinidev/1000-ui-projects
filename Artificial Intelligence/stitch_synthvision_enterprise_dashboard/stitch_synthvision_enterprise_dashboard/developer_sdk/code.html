import synth_vision as sv

client = sv.InferenceClient(
    api_key="YOUR_API_KEY",
    region="US_EAST_01"
)

# START OBJECT DETECTION STREAM
stream = client.stream_video(source="/dev/video0")

for detection in stream.detect():
    print(f"OBJECT: {detection.label} | CONF: {detection.score:.2%}")
    
    if detection.score > 0.98:
        client.trigger_alert(detection)
